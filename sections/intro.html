Quick Start Example

Here's a simple example to get you started with web scraping using Cheerio:

// Basic web scraping setup
const axios = require('axios');
const cheerio = require('cheerio');

// Function to scrape website titles
async function scrapeTitles() {
  try {
    const response = await axios.get('https://example.com');
    const $ = cheerio.load(response.data);
    
    // Extract all heading texts
    const titles = [];
    $('h1, h2, h3').each((index, element) => {
      titles.push($(element).text());
    });
    
    return titles;
  } catch (error) {
    console.error('Error scraping:', error);
  }
}

// Execute the scraper
scrapeTitles().then(titles => {
  console.log('Found titles:', titles);
});
